export const CreateToast = () => {
  const [addToast] = useStore((s) => [s.addToast])
  return (
    <Button
      onClick={() => {
        addToast({ children: Math.random(0, 100), id: Math.random(0, 100) })
      }}
    >
      Show Toast
    </Button>
  )
}
