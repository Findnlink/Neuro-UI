import react, { useState } from 'react'
import { Toggle, ComponentWrapper } from '../'

# Toggle

export const ToggleComponent = () => {
  const [value, setValue] = useState(false)
  return (
    <ComponentWrapper
      options={[
        { type: 'bool', name: 'primary', default: false },
        { type: 'bool', name: 'disabled', default: false },
        { type: 'bool', name: 'loading', default: false },
        { type: 'bool', name: 'error', default: false },
        {
          type: 'array',
          name: 'scale',
          default: 'm',
          values: ['s', 'm', 'l', 'xl']
        },
        {
          type: 'array',
          name: 'weight',
          default: 'medium',
          values: ['light', 'medium', 'bold']
        },
        {
          type: 'array',
          name: 'margin',
          default: 'm',
          values: ['0', 's', 'm', 'l', 'xl']
        },
        {
          type: 'array',
          name: 'padding',
          default: 's',
          values: ['0', 's', 'm', 'l', 'xl']
        }
      ]}
    >
      <Toggle onClick={() => setValue((prev) => !prev)} value={value} />
    </ComponentWrapper>
  )
}

<ToggleComponent />
